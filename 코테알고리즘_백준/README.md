def solution(n):
    numbers = [i for i in range(n+1)]

    start_idx, end_idx = 1, 1
    cnt = 1


    while end_idx != n:
        sum_ = sum(numbers[start_idx:end_idx+1])

        if sum_ == n:
            cnt += 1
            end_idx += 1
        elif sum_ < n:
            end_idx += 1
        else:
            start_idx += 1
    return cnt
