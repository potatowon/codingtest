def get_prime(num):
  prime = [False, False] + [True]*(num-1)
  for i in range(2, int(num**0.5)+1):
    if prime[i]:
      for j in range(i*2, num+1, i):
        prime[j] = False
        
  return [i for i in range(num+1) if prime[i]]
